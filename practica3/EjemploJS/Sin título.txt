            // C치lculo reutilizable

        // --- CONTROLADOR ---
        class CustomerController {
            // Simula traer datos de una DB (como MongoDB)

                ];

                // Convertimos los datos planos en OBJETOS de la clase Customer
                return rawData.map(d => new Customer(d.id, d.fullName, d.amount));
            }
        }

        // --- VISTA (Funci칩n que el bot칩n llama) ---
        async function loadTable() {
            const tbody = document.getElementById("tableBody");
            tbody.innerHTML = ""; // Limpiar tabla

            const customers = await CustomerController.fetchCustomers();

            customers.forEach(c => {
                const row = `<tr>
                    <td>${c.id}</td>
                    <td>${c.fullName}</td>
                    <td>${c.amount}</td>
                    <td>${c.calculateTotal().toFixed(2)}</td>
                </tr>`;
                tbody.innerHTML += row;
            });
        }
