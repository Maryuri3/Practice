// M√©todo para buscar por ID o Nombre
public Customer findCustomer(String searchTerm) {
    MongoCollection<Document> collection = DatabaseConnection.getDatabase().getCollection("customers");
    
    // Buscamos un documento donde el 'id' sea igual al t√©rmino (puedes cambiar a 'fullName')
    Document doc = collection.find(new Document("id", searchTerm)).first();

    if (doc != null) {
        return new Customer(
            doc.getString("id"),
            doc.getString("fullName"),
            doc.getDouble("amount")
        );
    }
    return null; // Si no lo encuentra
}
